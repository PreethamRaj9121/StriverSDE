
bool searchInBST(BinaryTreeNode<int> *root, int x) {
    // Write your code here.
    bool flag=false;
    
    while(root){
    if(root->data==x){
        return true;
    }
    else if(root->data>x){
        root=root->left;
    }
    else if (root->data<x){
        root=root->right;
    }
    }
    return false;
}
