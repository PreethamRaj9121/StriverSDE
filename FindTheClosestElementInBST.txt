class Solution
{
    public:
    //Function to find the least absolute difference between any node
	//value of the BST and the given integer.
    int minDiff(Node *root, int K)
    {
        //Your code here
        int res=INT_MAX;
        while(root){
            if(root->data==K){
                return 0;
            }
            else if(root->data > K){
                res=min(res,abs(root->data-K));
                root=root->left;
            }
            else{
                res=min(res,abs(root->data-K));
                root=root->right;
            }
        }
        return res;
    }
};
