#include <bits/stdc++.h> 
int lengthOfLongestConsecutiveSequence(vector<int> &arr, int n) {
    // Write your code here.
    int ans=0;
    unordered_set<int>s;
    
    for(int i=0;i<n;i++){
        s.insert(arr[i]);
    }
    for(int i=0;i<n;i++){
        if(s.find(arr[i]-1)!=s.end()){
            continue;
        }
        else{
            int count=0;
            int curr=arr[i];
            while(s.find(curr)!=s.end()){
                curr++;
                count++;
            }
            ans=max(ans,count);
        }
    }
    return ans;
}
