#include <bits/stdc++.h>

bool areAnagram(string &str1, string &str2){
    // Write your code here.
    
    int n1=str1.size();
    int n2=str2.size();
    
    if(n1!=n2) return false;
    
    unordered_map<char,int>mp;
    for(int i=0;i<n1;i++){
        mp[str1[i]]++; 
    }
    for(int i=0;i<n2;i++){
        if(mp.find(str2[i])==mp.end()){
            return false;
        }
    }
    return true;
    
}
