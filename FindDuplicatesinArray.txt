#include <bits/stdc++.h> 
int findDuplicate(vector<int> &arr, int n){
	// Write your code here.
    
    int ans=0;
    unordered_map<int,int>mp;
    for(int i=0;i<n;i++){
        mp[arr[i]]++;
    }
    for(auto x : mp){
        if(x.second>1){
            ans=x.first;
        }
    }
    return ans;
}
