#include <bits/stdc++.h> 
vector<int> ninjaAndSortedArrays(vector<int>& arr1, vector<int>& arr2, 
                                 int n, int m) {
	// Write your code here.
    
    int p1=n-1;
    int p2=m-1;
    int i=m+n-1;
    
    while(p2>=0){
        if(p1>=0 && arr1[p1]>arr2[p2]){
            arr1[i]=arr1[p1];
            i--;
            p1--;
        }
        else{
            arr1[i]=arr2[p2];
            i--;
            p2--;
        }
    }
    return arr1;
    
}
