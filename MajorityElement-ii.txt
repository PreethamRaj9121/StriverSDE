#include <bits/stdc++.h> 
vector<int> majorityElementII(vector<int> &arr)
{
    // Write your code here.
    vector<int>ans;
    int n=arr.size();
    int life1=0;
    int life2=0;
    int candidate1=INT_MIN;
    int candidate2=INT_MIN;
    
    for(int i=0;i<n;i++){
        if(candidate1==arr[i]){
            life1++;
        }
        else if(candidate2==arr[i]){
            life2++;
        }
        else if(life1==0){
            candidate1=arr[i];
            life1=1;
        }
        else if(life2==0){
            candidate2=arr[i];
            life2=1;
        }
        else{
            life1--;
            life2--;
        }
    }
    int count1=0;
    int count2=0;
    for(int i=0;i<n;i++){
        if(candidate1==arr[i]) count1++;
        if(candidate2==arr[i]) count2++;
    }
    if(count1>n/3) ans.push_back(candidate1);
    if(count2>n/3) ans.push_back(candidate2);
    
    return ans;

}
